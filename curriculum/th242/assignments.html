Assignment Overview

These assignments will guide you through the process of creating a multimedia project by building upon the provided SuperCollider sampler and Processing animation code snippets. Each assignment will help you develop and refine your skills in sound and media design.

Template Code Snippets

SuperCollider Template Code


// Boot the SuperCollider server
s.boot;

// Define the basic sampler function
(
{
    var freq = 440; // Frequency of the sound
    var amp = 0.5; // Amplitude of the sound
    var buf = Buffer.read(s, "path/to/sample.wav"); // Load the sample
    var snd = PlayBuf.ar(1, buf, BufRateScale.kr(buf), loop: 1, rate: freq, doneAction: 2);

    // Apply envelope to the sound
    var env = EnvGen.kr(Env.perc(0.01, 0.5), doneAction: 2);
    snd = snd * amp * env;

    // Apply basic effects
    snd = FreeVerb.ar(snd, 0.8, 0.9, 0.5); // Reverb effect

    Out.ar(0, snd);
}.play;
)

Processing Template Code


// Basic Animation with Interactivity in Processing
int circleColor;
float angle = 0;
float radius = 100;
float speed = 0.05;

void setup() {
    size(800, 600);
    circleColor = color(255, 0, 0); // Initial circle color
}

void draw() {
    background(0);
    
    // Update color based on mouse position
    float r = map(mouseX, 0, width, 0, 255);
    float g = map(mouseY, 0, height, 0, 255);
    circleColor = color(r, g, 255 - r);

    // Draw a rotating ellipse
    float x = width / 2 + cos(angle) * radius;
    float y = height / 2 + sin(angle) * radius;
    fill(circleColor);
    noStroke();
    ellipse(x, y, 50, 50);

    // Update angle for rotation
    angle += speed;

    // Draw a line from the center to the circle
    stroke(255);
    line(width / 2, height / 2, x, y);
}

// Adjust rotation speed with arrow keys
void keyPressed() {
    if (keyCode == LEFT) {
        speed -= 0.01;
    } else if (keyCode == RIGHT) {
        speed += 0.01;
    }
}
