Assignment 2: Simple Animation


int circleColor;
float angle = 0;
float radius = 100;
float speed = 0.05;

void setup() {
size(800, 600);
circleColor = color(255, 0, 0); // Initial color
}

void draw() {
background(0);

// Update color based on mouse position
float r = map(mouseX, 0, width, 0, 255);
float g = map(mouseY, 0, height, 0, 255);
circleColor = color(r, g, 255 - r);

// Draw rotating ellipse
float x = width / 2 + cos(angle) * radius;
float y = height / 2 + sin(angle) * radius;
fill(circleColor);
noStroke();
ellipse(x, y, 50, 50);

// Update angle
angle += speed;

// Draw line
stroke(255);
line(width / 2, height / 2, x, y);
}

void keyPressed() {
if (keyCode == LEFT) {
speed -= 0.01;
} else if (keyCode == RIGHT) {
speed += 0.01;
}
}

Assignment 3: Envelope Control


// Boot the SuperCollider server
s.boot;

// Sampler with Envelope Control
(
{
var freq = 440;
var amp = 0.5;
var buf = Buffer.read(s, "path/to/sample.wav");
var snd = PlayBuf.ar(1, buf, BufRateScale.kr(buf), loop: 1, rate: freq);

// Envelope control
var env = EnvGen.kr(Env.perc(0.01, 0.5), doneAction: 2);
snd = snd * amp * env;

// Effects
snd = FreeVerb.ar(snd, 0.8, 0.9, 0.5);

Out.ar(0, snd);
}.play;
)

Assignment 8: Audio-Visual Synchronization


int circleColor;
float angle = 0;
float radius = 100;
float speed = 0.05;

void setup() {
size(800, 600);
circleColor = color(255, 0, 0); // Initial color
}

void draw() {
background(0);

// Update color based on audio amplitude
float audioLevel = map(sin(angle), -1, 1, 0, 255); // Simulated audio level
circleColor = color(audioLevel, 255 - audioLevel, 255);

// Draw rotating ellipse
float x = width / 2 + cos(angle) * radius;
float y = height / 2 + sin(angle) * radius;
fill(circleColor);
noStroke();
ellipse(x, y, 50, 50);

// Update angle
angle += speed;

// Draw line
stroke(255);
line(width / 2, height / 2, x, y);
}
